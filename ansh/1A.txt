#include <stdio.h>
#include <stdlib.h>

int linearSearch(int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return i;
        }
    }
    return -1;
}

int main() {
    int rollNumbers[100];
    int numStudents = 0;
    int choice, rollToSearch, foundIndex;

    do {
        printf("\n--- Training Program Attendance Menu ---\n");
        printf("1. Add Student Roll Number\n");
        printf("2. Display All Attended Roll Numbers\n");
        printf("3. Search for a Student (Linear Search)\n");
        printf("4. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                if (numStudents < 100) {
                    printf("Enter roll number to add: ");
                    scanf("%d", &rollNumbers[numStudents]);
                    numStudents++;
                    printf("Roll number added successfully.\n");
                } else {
                    printf("Array is full. Cannot add more students.\n");
                }
                break;

            case 2:
                if (numStudents > 0) {
                    printf("Roll numbers of students who attended:\n");
                    for (int i = 0; i < numStudents; i++) {
                        printf("%d\n", rollNumbers[i]);
                    }
                } else {
                    printf("No students have attended yet.\n");
                }
                break;

            case 3:
                if (numStudents > 0) {
                    printf("Enter roll number to search: ");
                    scanf("%d", &rollToSearch);
                    foundIndex = linearSearch(rollNumbers, numStudents, rollToSearch);
                    if (foundIndex != -1) {
                        printf("Roll number %d attended the training program (found at index %d).\n", rollToSearch, foundIndex);
                    } else {
                        printf("Roll number %d did NOT attend the training program.\n", rollToSearch);
                    }
                } else {
                    printf("No students in the attendance list to search from.\n");
                }
                break;

            case 4:
                printf("Exiting program. Goodbye!\n");
                exit(0);

            default:
                printf("Invalid choice. Please try again.\n");
        }
    } while (choice != 4);

    return 0;
}
