#include <stdio.h>
#include <stdlib.h>
#define MAX 5

typedef struct {
    int orders[MAX];
    int front, rear;
} Queue;

void initQueue(Queue *q) {
    q->front = -1;
    q->rear = -1;
}

int isFull(Queue *q) {
    return (q->rear == MAX - 1);
}

int isEmpty(Queue *q) {
    return (q->front == -1 || q->front > q->rear);
}

void addJob(Queue *q, int jobID) {
    if (isFull(q)) {
        printf("Queue is full! Cannot accept more orders.\n");
        return;
    }

    if (q->front == -1) {
        q->front = 0;
    }
    q->orders[++(q->rear)] = jobID;
    printf("Order %d added to the queue.\n", jobID);
}

void deleteJob(Queue *q) {
    if (isEmpty(q)) {
        printf("Queue is empty! No orders to process.\n");
        return;
    }

    printf("Order %d processed and removed from the queue.\n", q->orders[q->front]);
    q->front++;
}

void displayQueue(Queue *q) {
    if (isEmpty(q)) {
        printf("Queue is empty!\n");
        return;
    }

    printf("Current Orders in Queue: ");
    for (int i = q->front; i <= q->rear; i++) {
        printf("%d ", q->orders[i]);
    }
    printf("\n");
}

int main() {
    Queue q;
    initQueue(&q);

    int choice, jobID;

    do {
        printf("\n--- Pizza Parlor Job Queue ---\n");
        printf("1. Add Job (New Order)\n");
        printf("2. Delete Job (Process Order)\n");
        printf("3. Display Queue\n");
        printf("4. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter Order ID: ");
                scanf("%d", &jobID);
                addJob(&q, jobID);
                break;
            case 2:
                deleteJob(&q);
                break;
            case 3:
                displayQueue(&q);
                break;
            case 4:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice! Try again.\n");
        }
    } while (choice != 4);

    return 0;
}
